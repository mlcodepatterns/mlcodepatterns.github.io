def _read_utf_8_file(filename):
    if sys.version_info.major == 2:  //// Python 2 specific
        with open(filename, "rb") as f:
            return unicode(f.read(), "utf-8")
    else:
        with open(filename, encoding="utf-8") as f:
            return f.read()
