    def _format_action(self, action):
        if action.dest == "help" or action.dest == "md":
            return ""
        lines = []
        lines.append("* **-%s %s** " % (action.dest,
                                        "[%s]" % action.default
                                        if action.default else ""))
        if action.help:
            help_text = self._expand_help(action)
            lines.extend(self._split_lines(help_text, 80))
