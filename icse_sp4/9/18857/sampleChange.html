def encrypt(input_string: str, key: int) -> str:
    result = ""
    for x in input_string:
        if not x.isalpha():
            result += x
        elif x.isupper():
            result += chr((ord(x) + key - 65) % 26 + 65)
        elif x.islower():
            result += chr((ord(x) + key - 97) % 26 + 97)
    return result
