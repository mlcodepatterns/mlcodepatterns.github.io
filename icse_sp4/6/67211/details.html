def cpu_load():
    with open("/proc/stat", "rt") as fp:
        for line in fp:
            line = line.strip()
            if line.startswith("cpu "):
                parts = list(map(int, line.split()[1:]))
                user_n = parts[0]
                sys_n = parts[2]
                idle_n = parts[3]
                tot = user_n + sys_n + idle_n
                return int(((user_n + sys_n) / tot) * 100)
    return 0
