def dot(a,b):
    dims = ndim(a) * ndim(b)
    if dims == 4:
        // matrix-matrix product
        return torch.mm(a, b)
    elif dims == 2:
        // matrix-vector product
        return torch.mv(a, b)
    elif dims == 1:
        // vector-vector product
        return torch.dot(a, b)
    else:
        raise ValueError("Cannot determine appropriate matrix product")
