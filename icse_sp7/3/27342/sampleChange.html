Before Change



def cumsum(x, axis=0):
    if axis is None:
        raise NotImplementedError("cumsum is not defined where axis is None")
    else:
        return torch.cumsum(x, dim=axis)

After Change



def cumsum(x, axis=None):
    if axis is None:
        return x.flatten().cumsum(dim=0)
    return torch.cumsum(x, dim=axis)
