    def turn_off(self, **kwargs):
        Turn the device off.
        set_req = self.gateway.const.SetReq

        value = kwargs.get("value")
        value_type = (

            if set_req.V_DIMMER in self._values else value_type)
        value = 0 if set_req.V_DIMMER in self._values else value
