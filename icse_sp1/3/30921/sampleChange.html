Before Change




def test_typechange(conf_scope):
    conf_scope({"a": "bar", "b": "foo", "c": 1})
    assert conf_scope.typechanges == {"a": (int, type("bar")),
                                      "b": (float, type("foo")),
                                      "c": (bool, int)}

def test_typechange(conf_scope):
    cfg = conf_scope({"a": "bar", "b": "foo", "c": 1})
    assert cfg.typechanges == {"a": (int, type("bar")),
                                      "b": (float, type("foo")),
                                      "c": (bool, int)}
