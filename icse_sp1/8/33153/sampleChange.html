def saveGraphToEdgeListTxt(graph, file_name):
    with open(file_name, "w") as f:
        f.write("%d\n" % len(graph.nodes))
        f.write("%d\n" % len(graph.edges))
        for i, j, w in graph.edges(data="weight", default=1):
            f.write("%d %d %f\n" % (i, j, w))
