def save_image_to_png(image_np, filename):
  from PIL import Image
  dirname = os.path.dirname(filename)
  if not os.path.exists(dirname):
    os.makedirs(dirname)
  img = Image.fromarray(np.uint8(image_np * 255.), "RGB")
  img.save(filename)
