def KernelMatrix(X, k=InnerProduct):
    n = X.size
    Q = zeros((n,n))
    for i in range(n):
       for j in range(n):
           // dumb, but how else to do outer products of arbitrary functions
           // without going through ufunc C-api ?
           Q[i,j] = k(X[i],X[j])
    return Q
