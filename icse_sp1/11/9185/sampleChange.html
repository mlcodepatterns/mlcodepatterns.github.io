def PrettyRelativeTime(time_diff_secs):
    secs = int(time_diff_secs)
    day = secs // (3600*24)
    hh = secs % (3600*24) // 3600
    mm = secs % 3600 // 60
    ss = secs % 60
    return "{}{:02d}:{:02d}:{:02d}".format("{} day{} ".format(day, "s" if day > 1 else "") if day > 0 else "", hh, mm, ss)
