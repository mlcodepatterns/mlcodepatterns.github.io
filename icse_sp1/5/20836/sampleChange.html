def _formats_for_docs(formats):
    lines = []
    for fmt in formats:
        fmts = _factor_format(fmt)
        if len(fmts) > 1:
            lines.append("- ``%s`` (%s)" % (
                fmts, ", ".join(":mod:`skbio.io.%s`" % s for s in fmts)))
        else:
            lines.append("- ``"%s"`` (:mod:`skbio.io.%s`)" % (fmt, fmt))
    return "\n".join(lines)
