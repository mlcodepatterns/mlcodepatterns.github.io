    @staticmethod
    def loadDictionary(fname):
        result = {}
        for lineNo, line in enumerate(open(fname)):
            pair = line[:-1].split("\t")
            if len(pair) != 2:
                continue
            wordId, word = pair
            result[int(wordId)] = word
        return result
