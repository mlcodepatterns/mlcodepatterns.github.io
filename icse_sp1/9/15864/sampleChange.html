    def __getitem__(self, index):

        img = Image.open(self.images[index]).convert("RGB")
        target = torch.load(self.targets[index])

        ow, oh = img.size

        // Reescale boxes
        target[:2].mul_(512 / ow)
        target[-2:].mul_(512 / oh)
