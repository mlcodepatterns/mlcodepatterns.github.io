    def run(self):
        print("running Dragon%i ..." % self.version)

                           (self.version, os.path.join(self.output_directory, self.drs_name)))
        if os_ret != 0:
            msg = "Oops, dragon%ishell command didn"t work! Are you sure Dragon%i software is installed on your machine?" % (
