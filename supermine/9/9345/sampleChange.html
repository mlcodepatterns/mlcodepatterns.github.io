def convertDateToDBFormat(string):
    valid_formats = ["%d-%m-%y", "%d/%m/%Y", "%d/%m/%y", "%Y-%m-%d"]
    result = None
    for valid_format in valid_formats:
        try:
            result = datetime.strptime(string, valid_format)
        except ValueError:
            continue
    return result
