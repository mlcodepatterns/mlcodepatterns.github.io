    text = []
    css_files = settings.COMPRESS_CSS["bookmarklet"]["source_filenames"]
    for filename in css_files:
        abs_filename = os.path.join(settings.MEDIA_ROOT, filename)
        f = open(abs_filename, "r")
        css = f.read()

        text.append(css)
