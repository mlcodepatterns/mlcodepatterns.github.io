def pad_filter(weight, old_size, new_size):
    pad_width = tuple(map(lambda x: (int(x), int(x)), np.subtract(np.array(new_size), np.array(old_size)) / 2))
    pad_width += ((0, 0), (0, 0))
    return np.pad(weight,
                  pad_width,
                  "constant",
                  constant_values=0)
