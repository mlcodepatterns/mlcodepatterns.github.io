    def get_artifacts(self, key):
        experiment = self.get_experiment(key, getinfo=False)
        if experiment.artifacts is None:
            return {}

        return {
            tag: self._get_file_url(art["key"]) for
            tag, art in experiment.artifacts.iteritems()
        }
