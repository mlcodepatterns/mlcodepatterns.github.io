    def get_columns(self, fields, rows):
        new_rows = []
        for row in rows:
            new_row = {}
            for field in fields:
                if field in row and row[field]:
                    new_row[field] = row[field]
            new_rows.append(new_row)
        return new_rows

    def get_columns(self, fields, rows):
        // Terse but efficient.
        new_rows = [{field: row[field] for field in fields if row.get(field, None)} for row in rows]
        return new_rows
