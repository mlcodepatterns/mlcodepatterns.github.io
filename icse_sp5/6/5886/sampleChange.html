    @built_in_intents.setter
    def built_in_intents(self, value):
        for intent in value:
            if not isinstance(intent, BuiltInIntent):
                raise ValueError("Expected a BuiltInIntent, found: %s"
                                 % type(intent))
        self._built_in_intents = value
        self.fitted = False
